## full-atom S-linked conjugation patches for cysteine
## This patch is used for S-linked glycosylation and ubiquitination.

NAME S-conjugated
TYPES SC_BRANCH_POINT

BEGIN_SELECTOR
PROPERTY PROTEIN
NOT VARIANT_TYPE SC_BRANCH_POINT
NOT VARIANT_TYPE BRANCH_LOWER_TERMINUS_VARIANT
NOT VARIANT_TYPE PHOSPHORYLATION
NOT VARIANT_TYPE SULFATION
NOT VARIANT_TYPE DEPROTONATED
END_SELECTOR


BEGIN_CASE  ## Cysteine #######################################################

END_SELECTOR

SET_ATOMIC_CHARGE HG 0.00

ADD_CONNECT SG ICOOR  -85.9775   68.374    1.439   SG   CB   CA

ADD_PROPERTY BRANCH_POINT

END_CASE
